def place(node, data, mode='', device_mode='', **kwargs):
    """Place an object using the robot arm and gripper.

    Args:
        - node (CustomNode): The ROS node with movement and grip agents.
        - data (dict): Task data with 'inputs' (format 'place::obj_name') and 'ins' (object info).
        - mode (str, optional): Mode suffix for skill agents. Defaults to ''.
        - device_mode (str, optional): Mode suffix for device agents. Defaults to ''.
        - **kwargs: Additional keyword arguments passed to `arm_approach` and `arm_back`.

    Returns:
        - dict: Updated `data` with 'isdone' (bool) set to True on success, or error result on failure.

    Raises:
        - KeyError: If 'ins', 'pose', or agent data is missing.
        - AttributeError: If agent data (e.g., `rev_data`) is unavailable.
    """
    lift_agent = node.agents[DEVICE_CLIENT_CONFIGS['lift']['agent_name'] + device_mode]
    armdmove_agent = node.agents[DEVICE_CLIENT_CONFIGS['arm_dmove']['agent_name'] + device_mode]
    grip_agent = node.agents[DEVICE_CLIENT_CONFIGS['grip']['agent_name'] + device_mode]
    movenangles_agent = node.agents[DEVICE_CLIENT_CONFIGS['move_angles']['agent_name'] + device_mode]
    liftstate_agent = node.agents[DEVICE_CLIENT_CONFIGS['lm_state']['agent_name'] + device_mode]
    robotpose_agent = node.agents[DEVICE_CLIENT_CONFIGS['robot_pose']['agent_name'] + device_mode]

    
    obj_name = data['inputs'].strip()
    if obj_name=='me':
        ret = armdmove_agent.send({'inputs': f'drz=40'})
        if not ret['isdone']:
            return ret

    else: 

        obj_lying = data['obj_lying'] if 'obj_lying' in data else True


        x0, y0, x1, y1, obj_depth = data['ins'][obj_name]['pose']
        xc, yc = (x0 + x1) // 2, (y0 + y1) // 2
        head_state = data['ins'][obj_name]['head_state']

        z, trans_armrobot_XYZ = calib.convert_femto_to_arm(
            xc, yc, obj_depth,
            liftstate_agent.rev_data['current_position'],
            robotpose_agent.rev_data['pose'],
            head_state['current_ry'],
            head_state['current_rz']
        )
        obj_center_height = data['obj_center_height'] if 'obj_center_height' in data else 0
        x, y = trans_armrobot_XYZ[:2]

        if obj_lying:
            z += 100

        z += CALIB_OFFSET[2]
        data['ins'][obj_name]['approach_pose'] = [x, y, z + obj_center_height + LIFTUP_Z]

        # Approach to center of object
        data['inputs'] = f'{obj_name}'
        ret = arm_approach(node=node, data=data, mode=mode, device_mode=device_mode, **kwargs)
        if not ret['isdone']:
            return ret
        
        if obj_lying and USE_LIFT:
            ret = dlift(node=node, data={'inputs': '-100'}, mode=mode, device_mode=device_mode, **kwargs)
            if not ret['isdone']:
                return ret

        # Object down
        if USE_LIFT:
            ret = dlift(node=node, data={'inputs': f'-{LIFTUP_Z}'}, mode=mode, device_mode=device_mode, **kwargs)
            if not ret['isdone']:
                return ret
        
        ret = armdmove_agent.send({'inputs': f'dz={-LIFTUP_Z}'})
        if not ret['isdone']:
            return ret
    
    # Grip open
    ret = grip_agent.send({'inputs': 'open'})
    if not ret['isdone']:
        return ret
    time.sleep(1.5)

    # arm up
    ret = armdmove_agent.send({'inputs': f'dz={LIFTUP_Z}'})
    if not ret['isdone']:
        return ret
    
    # Arm back
    ret = arm_back(node=node, device_mode=device_mode, **kwargs)
    if not ret['isdone']:
        return ret
    
    # Angle back
    # if obj_lying:
        # ret = armdmove_agent.send({'inputs': 'dry=-30'})
        # if not ret['isdone']:
        #     return ret
    ret = movenangles_agent.send({'inputs': 'ready'})
    if not ret['isdone']:
        return ret
        
    
    #Lift
    if USE_LIFT:
        ret = lift_agent.send({'inputs': 'home'})
        if not ret['isdone']:
            return ret
    
    data['isdone'] = True
    return data
