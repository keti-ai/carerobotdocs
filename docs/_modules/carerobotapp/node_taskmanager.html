def run_node():
    """ROS node for task manager

    - Initializes ROS node (manager) using CustomNode.
    - Configures task manager agent and agents for skills/devices per settings.
    - Processes task sequence (plan) by sending tasks to relevant agents.
    - Listens for incoming messages indefinitely.

    Args:
        None: This function takes no explicit arguments. It relies on configurations from `configs.tasks`
              (e.g., `MANAGER_CONFIGS`, `SKILL_CLIENT_CONFIGS`, `MODE`, `DEVICE_MODE`) and
              `configs.device_clients` (e.g., `DEVICE_CLIENT_CONFIGS`), parsed via `parse_keys_values`.

    Returns:
        None

    Raises:
        ImportError: If required modules (e.g., `pyconnect`, `configs.tasks`) are not installed.
        KeyError: If expected keys are missing from configuration dictionaries or agent responses.
        NotImplementedError: If a task action is not found in `SKILL_CLIENT_CONFIGS` or `DEVICE_CLIENT_CONFIGS`.
        AttributeError: If agent methods (e.g., `send`) or configurations are accessed incorrectly.

    Notes:
        - The `taskplan_responsefunc` processes a plan split into tasks (format: `action::param`).
    """
    from pyconnect.ros.custom_node import CustomNode
    from pyconnect.utils import str2dict, data_info
    from carerobotapp.configs.tasks import MANAGER_CONFIGS, SKILL_CLIENT_CONFIGS, MODE, DEVICE_MODE, DO_LOG_SUB, SHOW_NODE_CONFIGS, LLM_SERVES, DETECTOR_SERVERS
    from carerobotapp.configs.device_clients import DEVICE_CLIENT_CONFIGS
    from pyconnect.ros.utils import node_add_agents_from_configs, node_add_agent_from_config
    from carerobotapp.utils import init_llm_client, init_detect_client

    args = parse_keys_values(optional_args={'mode': MODE, 'devices_select': list(DEVICE_CLIENT_CONFIGS.keys()),
                                            'device_mode': DEVICE_MODE, 'skip_inactive': MANAGER_CONFIGS['skip_inactive']})

    # Deploy node
    node = CustomNode(name='manager', is_init_node=False, num_callbackgroup=4)
    node = node_add_agent_from_config(node=node, config=MANAGER_CONFIGS, mode=args['mode'])
    node = node_add_agents_from_configs(node=node, configs=SKILL_CLIENT_CONFIGS, mode=args['mode'])
    node = node_add_agents_from_configs(node=node, configs=DEVICE_CLIENT_CONFIGS, 
                                        mode=args['device_mode'], agents_select=args['devices_select'])
    node = node_add_agents_from_configs(node=node, configs=SHOW_NODE_CONFIGS)

    # Init LLM and VLA clients
    llm_name = MANAGER_CONFIGS["llm"]
    llm_client = init_llm_client(llm_name, LLM_SERVES[llm_name])

    if MANAGER_CONFIGS['prompt2']!='plan':
        openpi_client = init_detect_client(detect_url=DETECTOR_SERVERS[MANAGER_CONFIGS['prompt2']])

    # init logger
    if DO_LOG_SUB:
        device_node = CustomNode(name='device', is_init_node=False, num_callbackgroup=2)
        device_node = node_add_agents_from_configs(node=device_node, configs=DEVICE_CLIENT_CONFIGS, 
                                            mode=args['device_mode'], agents_select=args['devices_select'])
        logger = SubLog(node=device_node, every=0.2)
        device_node.spin(run_thread=True)        
    
    # get needed agents 
    findresult_agent = node.agents['find_result']
    femtorgb_agent = node.agents[args['device_mode'] + DEVICE_CLIENT_CONFIGS['femto_rgb']['agent_name']]
    dabairgb_agent = node.agents[args['device_mode'] + DEVICE_CLIENT_CONFIGS['dabai_rgb']['agent_name']]
    robotpose_agent = node.agents[args['device_mode'] + DEVICE_CLIENT_CONFIGS['robot_pose']['agent_name']]
    gripstate_agent = node.agents[args['device_mode'] + DEVICE_CLIENT_CONFIGS['grip_state']['agent_name']]
    

    def action2agentname(action):
        if action in SKILL_CLIENT_CONFIGS:
            return f'{args["mode"]}{action}'
        elif action in DEVICE_CLIENT_CONFIGS: 
            return f'{args["device_mode"]}{DEVICE_CLIENT_CONFIGS[action]["agent_name"]}'
        else:
            return None


    # Main task manager
    def exec_actions(actions):
        for j, plan in enumerate(actions):
            print(f'[{j+1}/{len(actions)}]: {plan}')
            tasks = [el for el in plan.split('\n') if len(el.split('::')) == 2]

            ts = []
            for i, task in enumerate(tasks):
                action, inputs = [el.lower().strip() for el in task.split('::')[:2]]

                if 'image_raw' in action:
                    if 'color' not in action:
                        continue
                    impath = json.loads(inputs.replace("'", '"'))['im']
                    for _ in range(10):
                        findresult_agent.send(cv2.imread(impath)[...,::-1])
                    
                    continue

                agent_name = action2agentname(action=action)
                # node.agents[agent_name].send({'inputs': inputs})
                data = {'inputs': inputs}
                ts.append(threading.Thread(target=node.agents[agent_name].send, args=(data,), daemon=True))
            for t in ts:
                t.start()
            for t in ts:
                t.join()
            
        
        return {'isdone': True}

    def exec_plan(plan, do_log_sub=False):
        tasks = [el for el in plan.split('\n') if len(el.split('::')) == 2]
        
        if do_log_sub:
            logger.start()
        
        prev_res = {'isdone': True}
        num_task = len(tasks)
        isdone = True
        for i, task in enumerate(tasks):
            print(f'[{i+1}/{num_task}]: {task}')
            action, inputs = [el.lower().strip() for el in task.split('::')[:2]]

            if not prev_res['isdone']:
                isdone = False
                break
            data = prev_res
            data['inputs'] = inputs


            agent_name = action2agentname(action=action)
            print(f'{agent_name} sending {data_info(data)}')
            
            # exec
            prev_res = node.agents[agent_name].send(data)
                
        if do_log_sub:
            logger.stop()
        
        return {'isdone': isdone}
    
    def actionchunk2actions(action_chunk):
        actions = []
        for state in action_chunk:
            plan = ''
            x,y,z = 1000*state[:3]
            plan += f'arm_move::{x:.2f},{y:.2f},{z:.2f}\n'

            gripper = state[-1]>0.5
            plan += f'grip::{1000 if gripper else 100}\n'

            actions.append(plan)
        return actions
            


    def exec_prompt(prompt, do_log_sub=False, prompt2='plan'):
        prompt2 = prompt2.strip().lower()
        if prompt2=='plan':
            if do_log_sub:
                logger.start(log_dir= '@' + prompt.replace(' ', '_'))
            plan = llm_client.chat_guide(prompt=prompt)
            print(plan)
            ret =  exec_plan(plan, do_log_sub=False)
            if do_log_sub:
                logger.stop()
        elif prompt2=='openpi':
            for _ in range(20):
                frame = {
                    'image': femtorgb_agent.rev_data['im'],
                    'wrist_image' : cv2.rotate(dabairgb_agent.rev_data['im'] , cv2.ROTATE_180),
                    'prompt': prompt}
                frame['state'] = (robotpose_agent.rev_data['pose']/1000.).tolist()
                frame['state'] = np.array(frame['state']  + [int(gripstate_agent.rev_data['motor_position']> 500), ])

                action_chunk = openpi_client.send({'frame': frame})
            
                ret = exec_actions(actions=actionchunk2actions(action_chunk=action_chunk))
        return ret
        
    
    def taskplan_responsefunc(data):
        if 'plan' in data:
            return exec_plan(plan=data['plan'], do_log_sub=DO_LOG_SUB)
        elif 'actions' in data:
            return exec_actions(actions=data['actions'])
        elif 'prompt' in data:
            exec_prompt(prompt=data['prompt'], do_log_sub=DO_LOG_SUB, prompt2=MANAGER_CONFIGS['prompt2'])
        else:
            return {'isdone': False}
        
    # Deploy node
    node.agents[MANAGER_CONFIGS['agent_name']].response_func = taskplan_responsefunc
    node.listen()


    
if __name__ == '__main__':
    run_node()
