def pick(node, data, mode='', device_mode='', **kwargs):
    """Pick an object using the robot arm and gripper.

    Args:
        - node (CustomNode): The ROS node with movement and grip agents.
        - data (dict): Task data with 'task' (format 'pick::obj_name') and 'ins' (object info).
        - mode (str, optional): Mode suffix for skill agents. Defaults to ''.
        - device_mode (str, optional): Mode suffix for device agents. Defaults to ''.
        - **kwargs: Additional keyword arguments passed to `arm_approach` and `arm_back`.

    Returns:
        - dict: Updated `data` with 'isdone' (bool) set to True on success, or error result on failure.

    Raises:
        - KeyError: If 'ins', 'pose', or agent data is missing.
        - AttributeError: If agent data (e.g., `rev_data`) is unavailable.
    """
    dmove_agent = node.agents[DEVICE_CLIENT_CONFIGS['dmove']['agent_name'] + device_mode]
    grip_agent = node.agents[DEVICE_CLIENT_CONFIGS['grip']['agent_name'] + device_mode]
    movenangles_agent = node.agents[DEVICE_CLIENT_CONFIGS['move_angles']['agent_name'] + device_mode]
    liftstate_agent = node.agents[DEVICE_CLIENT_CONFIGS['lm_state']['agent_name'] + device_mode]
    robotpose_agent = node.agents[DEVICE_CLIENT_CONFIGS['robot_pose']['agent_name'] + device_mode]
    lift_agent = node.agents[DEVICE_CLIENT_CONFIGS['lift']['agent_name'] + device_mode]

    ret = grip_agent.send({'task': 'grip::open::'})
    if not ret['isdone']:
        return ret
    
    # Approach to center of object
    obj_name = data['task'].split('::')[1].strip()
    x0, y0, x1, y1, obj_depth = data['ins'][obj_name]['pose']
    xc, yc = (x0 + x1) // 2, (y0 + y1) // 2
    head_state = data['ins'][obj_name]['head_state']

    z, trans_armrobot_XYZ = calib.convert_femto_to_arm(
        xc, yc, obj_depth,
        liftstate_agent.rev_data['current_position'] * 1000,
        robotpose_agent.rev_data['pose'],
        head_state['current_ry'],
        head_state['current_rz']
    )
    
    trans_lift_position, _ = calib.convert_femto_to_arm(
        x1, y1, obj_depth,
        liftstate_agent.rev_data['current_position'] * 1000,
        robotpose_agent.rev_data['pose'],
        head_state['current_ry'],
        head_state['current_rz']
    )
    x, y = trans_armrobot_XYZ[:2]
    data['ins'][obj_name]['approach_pose'] = [x, y, z]
    data['obj_center_height'] = z - trans_lift_position

    data['task'] = f'arm_aproach::{obj_name}::'
    ret = arm_approach(node=node, data=data, mode=mode, device_mode=device_mode, **kwargs)
    if not ret['isdone']:
        return ret
    
    # Grasp
    ret = grip_agent.send({'task': 'grip::150::'})
    if not ret['isdone']:
        return ret
    time.sleep(1.2)

    # Lift up
    ret = dmove_agent.send({'task': f'dmove::dz={LIFTUP_Z}::'})
    if not ret['isdone']:
        return ret

    # Arm back
    ret = arm_back(node=node, device_mode=device_mode, **kwargs)
    if not ret['isdone']:
        return ret
    
    ret = lift_agent.send({'task': 'lift::home::'})
    if not ret['isdone']:
        return ret
    
    ret = movenangles_agent.send({'task': 'move_angles::ready::'})
    if not ret['isdone']:
        return ret

    data['isdone'] = True
    return data
