def init_robot(node, device_mode='', **kwargs):
    """Initialize the robot to a ready state.

    Args:
        - node (CustomNode): The ROS node with device agents.
        - device_mode (str, optional): Mode suffix for agent names. Defaults to ''.
        - **kwargs: Additional keyword arguments (unused).

    Returns:
        - dict: Result with 'isdone' (bool) indicating success or failure.

    Raises:
        - KeyError: If agent configuration or data is missing.
        - AttributeError: If agent methods (e.g., `send`) are unavailable.
    """
    cleanerror_agent = node.agents[DEVICE_CLIENT_CONFIGS['clean_error']['agent_name'] + device_mode]
    setmode_agent = node.agents[DEVICE_CLIENT_CONFIGS['set_mode']['agent_name'] + device_mode]
    setstate_agent = node.agents[DEVICE_CLIENT_CONFIGS['set_state']['agent_name'] + device_mode]
    moveangles_agent = node.agents[DEVICE_CLIENT_CONFIGS['move_angles']['agent_name'] + device_mode]
    movehead = node.agents[DEVICE_CLIENT_CONFIGS['move_head']['agent_name'] + device_mode]

    try: 
        ret = cleanerror_agent.send({'task': 'clean_error::0::'})
        if not ret['isdone']:
            return ret

        ret = setmode_agent.send({'task': 'set_mode::0::'})
        if not ret['isdone']:
            return ret
        
        ret = setstate_agent.send({'task': 'set_state::0::'})
        if not ret['isdone']:
            return ret
        
        ret = moveangles_agent.send({'task': 'move_angles::home::'})
        if not ret['isdone']:
            return ret
        
        ret = moveangles_agent.send({'task': 'move_angles::ready::'})
        if not ret['isdone']:
            return ret
        
        ret = movehead.send({'task': 'move_head::ry=0::'})
        if not ret['isdone']:
            return ret
        
        ret = movehead.send({'task': 'move_head::ry=-25::'})
        if not ret['isdone']:
            return ret
        
        return {'isdone': True}
    except Exception as e:
        print(f'Init arm error: {e}')
        return {'isdone': False}

def arm_back(node, device_mode='', **kwargs):
    """Move the robot arm back to a predefined endpoint.

    Args:
        - node (CustomNode): The ROS node with arm movement agents.
        - device_mode (str, optional): Mode suffix for agent names. Defaults to ''.
        - **kwargs: Additional keyword arguments (unused).

    Returns:
        - dict: Result with 'isdone' (bool) indicating success or failure.

    Raises:
        - KeyError: If 'pose' data or agent config is missing.
        - AttributeError: If agent data (e.g., `rev_data`) is unavailable.
    """
    from configs.device_clients import ARM_ENPOINT
    
    armdmove_agent = node.agents[DEVICE_CLIENT_CONFIGS['arm_dmove']['agent_name'] + device_mode]
    robotpose_agent = node.agents[DEVICE_CLIENT_CONFIGS['robot_pose']['agent_name'] + device_mode]
    try:
        x, y = robotpose_agent.rev_data['pose'][:2]
        dx, dy = ARM_ENPOINT[0] - x, ARM_ENPOINT[1] - y
        ret = armdmove_agent.send({'task': f'arm_dmove::dx={dx},dy={dy}::'})
        if not ret['isdone']:
            return ret
    
        return {'isdone': True}
    except Exception as e:
        print(f'arm back error: {e}')
        return {'isdone': False}

def init_pick(node, device_mode='', **kwargs):
    """Prepare the robot for picking by setting initial pose and gripper state.

    Args:
        - node (CustomNode): The ROS node with pick-related agents.
        - device_mode (str, optional): Mode suffix for agent names. Defaults to ''.
        - **kwargs: Additional keyword arguments passed to `arm_back`.

    Returns:
        - dict: Result with 'isdone' (bool) indicating success or failure.

    Raises:
        - KeyError: If agent configuration or data is missing.
        - AttributeError: If agent methods (e.g., `send`) are unavailable.
    """
    grip_agent = node.agents[DEVICE_CLIENT_CONFIGS['grip']['agent_name'] + device_mode]
    lift_agent = node.agents[DEVICE_CLIENT_CONFIGS['lift']['agent_name'] + device_mode]
    moveangles_agent = node.agents[DEVICE_CLIENT_CONFIGS['move_angles']['agent_name'] + device_mode]
    movehead = node.agents[DEVICE_CLIENT_CONFIGS['move_head']['agent_name'] + device_mode]
    try:
        # Go to init pose
        ret = movehead.send({'task': 'move_head::ry=-25::'})
        if not ret['isdone']:
            return ret
        
        ret = arm_back(node=node, device_mode=device_mode, **kwargs)
        if not ret['isdone']:
            return ret
        
        ret = grip_agent.send({'task': 'grip::open::'})
        if not ret['isdone']:
            return ret
        
        ret = moveangles_agent.send({'task': 'move_angles::ready::'})
        if not ret['isdone']:
            return ret
        
        ret = lift_agent.send({'task': 'lift::home::'})
        if not ret['isdone']:
            return ret
    
        return {'isdone': True}
    except Exception as e:
        print(f'Init pick error: {e}')
        return {'isdone': False}
