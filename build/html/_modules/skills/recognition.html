def find_surface(rgb, depth, obj_name='top surface'):
    """Detect a surface in RGB and depth images.

    Args:
        - rgb (np.ndarray): RGB image array.
        - depth (np.ndarray): Depth image array.
        - obj_name (str, optional): Name of the object to detect. Defaults to 'top surface'.

    Returns:
        - dict: Result with 'isdone' (bool), 'plane_coeffs' (list), and 'im' (np.ndarray) if successful.

    Raises:
        - AttributeError: If `detect_client` or `llm_client` methods fail.
    """
    ret = detect_client.send({'rgb': rgb, 'caption': obj_name})
    if ret is None:
        return None
    pred, out = ret
    num_pred = 0 if pred is None else len(pred["boxes"])
    print(f'{num_pred} {obj_name} detected')

    llm_ret = llm_client.chat_findobj(rgb=rgb, obj_name=obj_name)
    print(f'LLM Verify: {llm_ret} {obj_name} detected')
    
    num_detected = min(num_pred, llm_ret)
    issucced = num_detected > 0
    print(f'Finally: {num_detected} {obj_name} detected.')

    coeffs = get_masked_depthplane(depth=depth, mask=pred['masks'][0]) if issucced else None
    return {'isdone': issucced, 'plane_coeffs': coeffs, 'im': out if issucced else rgb}

def detect(node, data, show_mode='', device_mode='', camera=CONFIGS['camera'], **kwargs):
    """Detect objects using camera data.

    Args:
        - node (CustomNode): The ROS node with camera and result agents.
        - data (dict): Task data with 'task' (format 'detect::obj_name').
        - show_mode (str, optional): Mode suffix for show agents. Defaults to ''.
        - device_mode (str, optional): Mode suffix for device agents. Defaults to ''.
        - camera (str, optional): Camera type from CONFIGS. Defaults to CONFIGS['camera'].
        - **kwargs: Additional keyword arguments (unused).

    Returns:
        - dict: Result with 'isdone' (bool), 'ins' (dict of object info), and 'head_state'.

    Raises:
        - KeyError: If agent data or 'task' format is invalid.
    """
    findresult_agent = node.agents[f'find_result{show_mode}']
    findresult_agent.do_log_info = False

    headstate_agent = node.agents[f"{DEVICE_CLIENT_CONFIGS['head_state']['agent_name']}{device_mode}"]
    rgb_agent = node.agents[f"{DEVICE_CLIENT_CONFIGS[f'{camera}_rgb']['agent_name']}{device_mode}"]
    depth_agent = node.agents[f"{DEVICE_CLIENT_CONFIGS[f'{camera}_depth']['agent_name']}{device_mode}"]
    caminfo_agent = node.agents[f"{DEVICE_CLIENT_CONFIGS[f'{camera}_caminfo']['agent_name']}{device_mode}"]

    rgb = rgb_agent.rev_data['im']
    depth = depth_agent.rev_data['im']
    cam_params = caminfo_agent.rev_data['cam_params']
    head_state = headstate_agent.rev_data
    _, inp_name = data['task'].split('::')[:2]
    inp_name = inp_name.replace(',', '.')
    objnames = [el.strip() for el in inp_name.split(',')] if ',' in inp_name else [inp_name]

    ins_out = {objname: {'score': -1, 'pose': None, 'head_state': None} for objname in objnames}
    out = {'isdone': False, 'ins': ins_out, 'head_state': head_state}
    rgb_out = rgb.copy()
    if 'me' in inp_name:
        out.update({'isdone': True})
    else:
        ins = find_client.send({'rgb': rgb, 'min_mass': 2000, 'max_ratio': 0.6, 'caption': inp_name})
        print(data_info(ins.todict()))
        if len(ins) > 0:
            valid_depth = depth > 0
            for objname in objnames:
                if objname not in ins.label_clusters:
                    continue
                v = ins.label_clusters[objname]
                depth_value = np.median(depth[valid_depth & (v.masks[0] > 0)])
                x0, y0, x1, y1 = v.boxes[0, :]
                ins_out[objname]['pose'] = [x0, y0, x1, y1, depth_value]
                ins_out[objname]['score'] = v.scores.flatten()[0]
                rgb_out = v.visualize_normalvectors(rgb=rgb_out, depth=depth, cam_params=cam_params, target_ind=0)

            out['isdone'], out['ins'] = True, ins_out
            rgb_out = ins.visualize(rgb=rgb)

    for _ in range(10):
        findresult_agent.send(rgb_out)
    findresult_agent.log_msg({'rgb': rgb, 'depth': depth, 'caminfo': cam_params, 'result': rgb_out}, msg_type='returned')
    
    return out

def find(node, data, show_mode='', device_mode='', **kwargs):
    """Find objects by scanning with head movement.

    Args:
        - node (CustomNode): The ROS node with movement and detection agents.
        - data (dict): Task data with 'task' (format 'find::obj_name').
        - show_mode (str, optional): Mode suffix for show agents. Defaults to ''.
        - device_mode (str, optional): Mode suffix for device agents. Defaults to ''.
        - **kwargs: Additional keyword arguments passed to `detect`.

    Returns:
        - dict: Result with 'isdone' (bool) and 'ins' (dict of object info).

    Raises:
        - KeyError: If agent data or 'task' format is invalid.
    """
    movehead_agent = node.agents[DEVICE_CLIENT_CONFIGS['move_head']['agent_name'] + device_mode]

    inp = data['task'].split('::')[1].strip()
    objnames = [el.strip() for el in inp.split(',')] if ',' in inp else [inp]

    isdone = False
    ins = {objname: {'score': -1, 'pose': None, 'head_state': None} for objname in objnames}
    for ry in [0, -40]:
        ret = movehead_agent.send({'task': f'move_head::ry={ry}::'})
        if not ret['isdone']:
            return ret
        
        ret = detect(node=node, data=data, show_mode=show_mode, device_mode=device_mode, **kwargs)
        if not ret['isdone']:
            continue
        
        for objname in objnames:
            if objname not in ret['ins']:
                continue
            v = ret['ins'][objname]
            score = v['score']
            if score > ins[objname]['score']:
                ins[objname]['pose'] = v['pose']
                ins[objname]['score'], ins[objname]['head_state'] = score, ret['head_state']
                isdone = True

    ret = movehead_agent.send({'task': f'move_head::ry=-25::'})
    if not ret['isdone']:
        return ret
    
    return {'isdone': isdone, 'ins': ins}

def find_tooltip(node, data, show_mode='', device_mode='', camera=findtooltip_cfgs['camera'], **kwargs):
    """Locate tooltips by moving and detecting.

    Args:
        - node (CustomNode): The ROS node with movement and camera agents.
        - data (dict): Task data (unused here).
        - show_mode (str, optional): Mode suffix for show agents. Defaults to ''.
        - device_mode (str, optional): Mode suffix for device agents. Defaults to ''.
        - camera (str, optional): Camera type from config. Defaults to findtooltip_cfgs['camera'].
        - **kwargs: Additional keyword arguments (unused).

    Returns:
        - dict: Result with 'isdone' (bool) and 'tooltip_mask' (np.ndarray).

    Raises:
        - KeyError: If agent data is missing.
    """
    findresult_agent = node.agents[f'find_result{show_mode}']
    findresult_agent.do_log_info = False
    rgb_agent = node.agents[f"{DEVICE_CLIENT_CONFIGS[f'{camera}_rgb']['agent_name']}{device_mode}"]

    dmove_agent = node.agents[DEVICE_CLIENT_CONFIGS['dmove']['agent_name'] + device_mode]
    moveangles_agent = node.agents[DEVICE_CLIENT_CONFIGS['move_angles']['agent_name'] + device_mode]

    ret = moveangles_agent.send({'task': 'move_angles::ready::'})
    if not ret['isdone']:
        return ret
    time.sleep(1)
    ret0 = detect_tooltip(node=node, data=data, show_mode=show_mode, device_mode=device_mode, camera=camera)

    ret = dmove_agent.send({'task': 'dmove::dx=20,dy=20,dz=20::'})
    if not ret['isdone']:
        return ret
    time.sleep(1)
    ret1 = detect_tooltip(node=node, data=data, show_mode=show_mode, device_mode=device_mode, camera=camera)

    tooltips_mask = get_common_mask(ins2masks(ret0['ins']), ins2masks(ret1['ins']))
    rgb_out = show_masks_on_rgb(rgb_agent.rev_data['im'], [tooltips_mask])
    for _ in range(10):
        findresult_agent.send(rgb_out)

    ret = moveangles_agent.send({'task': 'move_angles::ready::'})
    if not ret['isdone']:
        return ret

    return {'tooltip_mask': tooltips_mask, 'isdone': True}

def track(node, data, show_mode='', device_mode='', camera=track_cfgs['camera'], **kwargs):
    """Track an object and align tooltip with its center.

    Args:
        - node (CustomNode): The ROS node with movement and camera agents.
        - data (dict): Task data with 'task' (format 'track::obj_name').
        - show_mode (str, optional): Mode suffix for show agents. Defaults to ''.
        - device_mode (str, optional): Mode suffix for device agents. Defaults to ''.
        - camera (str, optional): Camera type from config. Defaults to track_cfgs['camera'].
        - **kwargs: Additional keyword arguments (unused).

    Returns:
        - dict: Result with 'isdone' (bool) indicating success or failure.

    Raises:
        - KeyError: If agent data or 'task' format is invalid.
    """
    findresult_agent = node.agents[f'find_result{show_mode}']
    findresult_agent.do_log_info = False

    rgb_agent = node.agents[f"{DEVICE_CLIENT_CONFIGS[f'{camera}_rgb']['agent_name']}{device_mode}"]
    depth_agent = node.agents[f"{DEVICE_CLIENT_CONFIGS[f'{camera}_depth']['agent_name']}{device_mode}"]
    caminfo_agent = node.agents[f"{DEVICE_CLIENT_CONFIGS[f'{camera}_caminfo']['agent_name']}{device_mode}"]

    dmove_agent = node.agents[DEVICE_CLIENT_CONFIGS['dmove']['agent_name'] + device_mode]
    moveangles_agent = node.agents[DEVICE_CLIENT_CONFIGS['move_angles']['agent_name'] + device_mode]

    obj_name = data['task'].split('::')[1]
    ret = detect_func(rgb=rgb_agent.rev_data['im'], depth=depth_agent.rev_data['im'], detect_client=find_client,
                      caption=obj_name, head_state=None, cam_params=caminfo_agent.rev_data['cam_params'])
    obj_mask = ret[0]['ins'][obj_name]['masks'][0]
    obj_box = ret[0]['ins'][obj_name]['boxes'][0, :]
    obj_center = (obj_box[0] + obj_box[2]) // 2, (obj_box[1] + obj_box[3]) // 2
    
    h, w = rgb_agent.rev_data['im'].shape[:2]
    dd = w // 2 - obj_center[0]

    ret0 = track_client.send({'rgb': rgb_agent.rev_data['im'], 'min_mass': 2000, 'max_ratio': 0.5})
    if ret0 is None:
        return {'isdone': False}

    while abs(dd) > 10:
        dy = 20 if dd > 0 else -20
        ret = dmove_agent.send({'task': f'dmove::dy={dy}::'})
        if not ret['isdone']:
            return ret
        ret1 = findtooltip_client.send({'rgb': rgb_agent.rev_data['im'], 'min_mass': 2000, 'max_ratio': 0.5})
        if ret1 is None:
            return {'isdone': False}
        tooltip_mask = get_common_mask(ins2masks(ret0[0]), ins2masks(ret1[0]))
        Y, X = np.where(tooltip_mask)
        tooltip_box = [np.amin(X), np.amin(Y), np.amax(X), np.amax(Y)]
        tooltip_center = (tooltip_box[0] + tooltip_box[2]) // 2, (tooltip_box[1] + tooltip_box[3]) // 2

        ret = detect_func(rgb=rgb_agent.rev_data['im'], depth=depth_agent.rev_data['im'], detect_client=find_client,
                          caption=obj_name, head_state=None, cam_params=caminfo_agent.rev_data['cam_params'])
        masks, boxes = ins2masks(ret[0]['ins']), ins2boxes(ret[0]['ins'])

        inners = [np.sum(np.bitwise_and(obj_mask, m)) for m in masks]
        outers = [np.sum(np.bitwise_or(obj_mask, m)) for m in masks]
        ious = [inner / outer for inner, outer in zip(inners, outers)]
        argmax = int(np.argmax(ious))
        
        obj_mask, obj_box = masks[argmax], boxes[argmax, :].flatten()
        obj_center = (obj_box[0] + obj_box[2]) // 2, (obj_box[1] + obj_box[3]) // 2
        dd = tooltip_center[0] - obj_center[0]
        rgb_out = show_masks_on_rgb(rgb_agent.rev_data['im'], [obj_mask, tooltip_mask])
        for _ in range(10):
            findresult_agent.send(rgb_out)        

    return {'isdone': True}

    cam_params = caminfo_agent.rev_data['cam_params']
    head_state = headstate_agent.rev_data
    _, inp_name = data['task'].split('::')[:2]
