def get_actions_from_trajectory(trajectory=None):
    states = get_states_from_trajectory(trajectory=trajectory)
    plans = []
    for state in states:
        plan = ''
        plan += state2liftprompt(state=state)
        plan += state2gripprompt(state=state)
        plan += state2armmoveprompt(state=state)
        plan += state2imagepaths(state=state)

        plans.append(plan)

    return plans

def create_plane(size=10, step=1, noise_level=0.1):
    x = np.arange(-size, size, step)
    y = np.arange(-size, size, step)
    xx, yy = np.meshgrid(x, y)
    zz = np.zeros_like(xx) + np.random.normal(0, noise_level, xx.shape)
    points = np.vstack((xx.ravel(), yy.ravel(), zz.ravel())).T
    return points

def create_parabolic(size=10, step=1, noise_level=0.1):
    x = np.arange(-size, size, step)
    y = np.arange(-size, size, step)
    xx, yy = np.meshgrid(x, y)
    zz = 0.1 * (xx**2 + yy**2) + np.random.normal(0, noise_level, xx.shape)
    points = np.vstack((xx.ravel(), yy.ravel(), zz.ravel())).T
    return points

def calc_normalvector(points):
    centroid = np.mean(points, axis=0)
    centered_points = points - centroid
    cov_matrix = np.cov(centered_points, rowvar=False)
    _, _, vh = np.linalg.svd(cov_matrix)
    normal_vector = vh[-1]  # The last singular vector is the normal
    return normal_vector

def visualize_normalvector(points, normal_vector=None):
    # Create Open3D point cloud
    pcd = o3d.geometry.PointCloud()
    pcd.points = o3d.utility.Vector3dVector(points)
    surfaces = [pcd,]

    # Create line set to represent normal vector
    if normal_vector is not None:
        centroid = np.mean(points, axis=0)
        line_points = [centroid, centroid + 10*normal_vector]
        lines = [[0, 1]]
        line_set = o3d.geometry.LineSet()
        line_set.points = o3d.utility.Vector3dVector(line_points)
        line_set.lines = o3d.utility.Vector2iVector(lines)
        surfaces.append(line_set)

    # Visualize
    o3d.visualization.draw_geometries(surfaces)

def visualize_pointclouds(pointclouds, colors=None):

        pcd.points = o3d.utility.Vector3dVector(points)

            pcd.colors = o3d.utility.Vector3dVector(colors[i])
        surfaces.append(pcd)

    # Visualize
    o3d.visualization.draw_geometries(surfaces)

def visualize_rgbd_pointcloud(depth, cam_params, rgb=None, normal_point=None, normal_vector=None):
    fx, fy, cx, cy = cam_params
    h, w = depth.shape
    
    # Generate 3D points
    x, y = np.meshgrid(np.arange(w), np.arange(h))
    x3 = (x - cx) * depth / fx
    y3 = (y - cy) * depth / fy
    z3 = depth
    points = np.stack((x3, y3, z3), axis=-1).reshape(-1, 3)
    
    # Create Open3D point cloud
    pcd = o3d.geometry.PointCloud()
    pcd.points = o3d.utility.Vector3dVector(points)
    
    # Assign RGB values as color
    if rgb is not None:
        colors = rgb.reshape(-1, 3) / 255.0
        pcd.colors = o3d.utility.Vector3dVector(colors)

    geometries = [pcd]

    # Add normal vector visualization if provided
    if normal_point is not None and normal_vector is not None:
        line_points = [normal_point, normal_point + 50*normal_vector]
        lines = [[0, 1]]
        line_set = o3d.geometry.LineSet()
        line_set.points = o3d.utility.Vector3dVector(line_points)
        line_set.lines = o3d.utility.Vector2iVector(lines)
        geometries.append(line_set)
    
    # Visualize
    o3d.visualization.draw_geometries(geometries)

def xyz2Ixy(x,y,z, cam_params, eps=1e-10):
    fx, fy, xc, yc = cam_params[:4]
    Ix =  np.divide(x, z + eps) * fx + xc
    Iy =  np.divide(y, z + eps) * fy + yc
    return int(Ix), int(Iy)

def Ixy2xyz(Ix, Iy, Z, cam_params):
    fx, fy, xc, yc = cam_params[:4]
    Ix, Iy = Ix - xc, Iy - yc
    return np.multiply(Ix, Z) / fx, np.multiply(Iy, Z) / fy, Z

    def load(self):
        traj_dirs = sorted(list(glob.glob(os.path.join(self.data_dir, '*'))))
        num_traj = len(traj_dirs)
        
        self.episodes =  []
        for i in range(0, num_traj):
            traj_dir = traj_dirs[i]
            # print(f'[{i+1}/{num_traj}] {traj_dir}')
            traj_dir, traj_name = os.path.split(traj_dir)
            steps = get_states_from_trajectory(trajectory=traj_name, log_dir=traj_dir)
            prompt, steps = steps[0]['prompt'], steps[1:]
            num_step = len(steps)


            episode = {'prompt': prompt, 'images': [], 'wrist_images':[], 
                       'states':[], 'grippers':[], 'next_states':[], 'next_grippers': []}
            for j in range(self.start_from, num_step-self.stride, self.stride):
                step_data, next_step_data = steps[j], steps[j+self.stride]

                image = step_data['/femto/color/image_raw/compressed']['im']
                wrist_image = step_data['/hand/color/image_raw']['im']

                robot_state = step_data['/xarm/robot_states']['pose']
                gripper_state = step_data['/gripper/state']['motor_position'] 

                next_robot_state = next_step_data['/xarm/robot_states']['pose']
                next_gripper_state = next_step_data['/gripper/state']['motor_position'] 
                

                episode['images'].append(image)
                episode['wrist_images'].append(wrist_image)
                episode['states'].append(robot_state)
                episode['grippers'].append(gripper_state)
                episode['next_states'].append(next_robot_state)
                episode['next_grippers'].append(next_gripper_state)


            self.episodes.append(episode)



    def __len__(self):
        return len(self.episodes)

    def get_frame(self, episode_ind=0, frame_ind=0):
        episode = self.episodes[episode_ind]
        frame = {'prompt': episode['prompt']}
        frame['image'] = cv2.imread(episode['images'][frame_ind])[...,::-1]
        frame['wrist_image'] = cv2.imread(episode['wrist_images'][frame_ind])[...,::-1]
        frame['wrist_image'] = cv2.rotate(frame['wrist_image'] , cv2.ROTATE_180)


        frame['state'] = (np.load(episode['states'][frame_ind])/1000.).tolist()
        frame['state'] = np.array(frame['state']  + [int(episode['grippers'][frame_ind] > 500), ])

        frame['actions'] = (np.load(episode['next_states'][frame_ind])/1000.).tolist()
        frame['actions'] = np.array(frame['actions']  + [int(episode['grippers'][frame_ind] > 500), ])


        return frame

    def log_episode(self, episode_ind=0, fps=5, log_dir=''):
        episode = self.episodes[episode_ind]
        video_path = os.path.join(log_dir, f'episode{episode_ind}_{episode["prompt"]}.mp4')

        frame0 = self.get_frame(episode_ind=episode_ind, frame_ind=0)
        h,w = frame0['wrist_image'].shape[:2]
        
        out = cv2.VideoWriter(video_path, cv2.VideoWriter_fourcc(*'mp4v'), fps, (2*w, h+h//2))

        for i in range(len(episode['states'])):
            print(f'Writing frame {i}')
            frame = self.get_frame(episode_ind=episode_ind, frame_ind=i)
            image, wrist_image = frame['image'], frame['wrist_image']
            h, w = wrist_image.shape[:2]
            image = cv2.resize(image,dsize=(w,h), interpolation=cv2.INTER_CUBIC)


            white_image = 255*np.ones((h//2, 2*w,3), 'uint8')
            cv2.putText(white_image, f'Frame {i}: {frame["prompt"]}', (20, 60), cv2.FONT_HERSHEY_COMPLEX, 0.8, (0,0,0), 1)
            cv2.putText(white_image, f'state: {frame["state"]}', (20, 100), cv2.FONT_HERSHEY_COMPLEX, 0.8, (0,0,0), 1)
            cv2.putText(white_image, f'actions: {frame["actions"]}', (20, 140), cv2.FONT_HERSHEY_COMPLEX, 0.8, (0,0,0), 1)
            
            concat = np.concatenate((np.concatenate((image, wrist_image), axis=1), white_image), axis=0)
            out.write(concat[...,::-1])
        
        out.release()
        print(f'Episode {episode_ind} written in {video_path}')
