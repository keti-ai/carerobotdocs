def run_node():
    """ROS node for task manager

    - Initializes ROS node (manager) using CustomNode.
    - Configures task manager agent and agents for skills/devices per settings.
    - Processes task sequence (plan) by sending tasks to relevant agents.
    - Listens for incoming messages indefinitely.

    Args:
        None: This function takes no explicit arguments. It relies on configurations from `configs.tasks`
              (e.g., `MANAGER_CONFIGS`, `SKILL_CLIENT_CONFIGS`, `MODE`, `DEVICE_MODE`) and
              `configs.device_clients` (e.g., `DEVICE_CLIENT_REQUEST_CONFIGS`), parsed via `parse_keys_values`.

    Returns:
        None

    Raises:
        ImportError: If required modules (e.g., `pyconnect`, `configs.tasks`) are not installed.
        KeyError: If expected keys are missing from configuration dictionaries or agent responses.
        NotImplementedError: If a task action is not found in `SKILL_CLIENT_CONFIGS` or `DEVICE_CLIENT_REQUEST_CONFIGS`.
        AttributeError: If agent methods (e.g., `send`) or configurations are accessed incorrectly.

    Notes:
        - The `taskplan_responsefunc` processes a plan split into tasks (format: `action::param`).
    """
    import time
    from pyconnect.ros.custom_node import CustomNode
    from pyconnect.utils import str2dict, data_info
    from carerobotapp.configs.tasks import MANAGER_CONFIGS, SKILL_SERVER_CONFIGS, MODE, DEVICE_MODE, DO_LOG_SUB, SHOW_NODE_CONFIGS, LLM_SERVES, DETECTOR_SERVERS
    from pyconnect.ros.utils import node_add_agents_from_configs, node_add_agent_from_config
    from carerobotapp.utils import init_llm_client, init_detect_client

    args = parse_keys_values(optional_args={'mode': MODE, 'devices_select': list(DEVICE_CLIENT_REQUEST_CONFIGS.keys()),
                                            'device_mode': DEVICE_MODE, 'skip_inactive': MANAGER_CONFIGS['skip_inactive']})

    # Deploy node
    node = CustomNode(name='manager', is_init_node=False, num_callbackgroup=4)
    node = node_add_agent_from_config(node=node, config=MANAGER_CONFIGS, mode=args['mode'])
    # node = node_add_agents_from_configs(node=node, configs=SKILL_SERVER_CONFIGS, mode=args['mode'])
    
    node = node_add_agents_from_configs(node=node, configs=SHOW_NODE_CONFIGS)
    node = node_add_agents_from_configs(node=node, configs=DEVICE_CLIENT_REQUEST_CONFIGS, 
                                        mode=args['device_mode'], agents_select=args['devices_select'])  


    # Init LLM and VLA clients
    llm_name = MANAGER_CONFIGS["llm"]
    llm_client = init_llm_client(llm_name, LLM_SERVES[llm_name])

    if MANAGER_CONFIGS['prompt2']!='plan':
        if 'openpi' in MANAGER_CONFIGS['prompt2']:
            openpi_client = init_detect_client(detect_url=DETECTOR_SERVERS['openpi'])
        else:
            NotImplementedError

    # init logger
    if DO_LOG_SUB:
        # node = CustomNode(name='device', is_init_node=False, num_callbackgroup=2)
        # node = node_add_agents_from_configs(node=node, configs=DEVICE_CLIENT_REQUEST_CONFIGS, 
        #                                     mode=args['device_mode'], agents_select=args['devices_select'])
        logger = SubLog(node=node, every=0., agents_select=[
            'femto_rgb', 'femto_depth', 'femto_caminfo',
            'dabai_rgb', 'dabai_depth', 'dabai_caminfo',
            'grip_state', 'robot_pose', 'lm_state', 'head_state'
        ])
        # node.spin(run_thread=True)

    def refine_plan(plan):
        tasks = [task.replace('\n','') for task in plan.split('\n') if len(task.split('::'))==2]

        refined_tasks = []
        num_task = len(tasks)
        for i,task in enumerate(tasks):
            action, inputs = task.split('::')
            inputs = 'inputs=None' if len(inputs.strip())==0 else inputs.strip()
            if '=' not in inputs:
                inputs = evaluate(inputs)
                inputs = f'inputs="{inputs}"' if isinstance(inputs, str) else f'inputs={inputs}'
            refined_tasks.append(f'print(\'{">"*20}[{i+1}/{num_task}] {action}({inputs}){"<"*20}\')')
            refined_tasks.append(f'assert prev_res["isdone"]')
            refined_tasks.append(f'params.update(prev_res)')
            # refined_tasks.append(f'params.pop("inputs", None)')
            refined_tasks.append(f'prev_res.update({action}({inputs}, **params))')
        return '\n'.join(refined_tasks)


    # Main task manager
    def exec_actions(actions):
        # common inputs

        if not isinstance(actions, (list, tuple)):
            actions = [actions,]
        for j, plan in enumerate(actions):
            print(f'[{j+1}/{len(actions)}]: {plan}')
            tasks = [el for el in plan.split('\n') if len(el.split('::')) == 2]

            ts = []
            for i, task in enumerate(tasks):
                action, inputs = [el.lower().strip() for el in task.split('::')[:2]]
                inputs  = inputs if '=' in inputs else f'inputs="{inputs}"'
                if 'image_raw' in action:
                    if 'color' not in action:
                        continue
                    impath = json.loads(inputs.replace("'", '"'))['im']
                    for _ in range(10):
                        node.agents[args['mode']+'find_result'].send(cv2.imread(impath)[...,::-1])
                    
                    continue

                def func():
                    params = {'node': node}
                    return exec(f'{action}({inputs}, **params)')

                ts.append(threading.Thread(target=func, daemon=True))
            for t in ts:
                t.start()
            for t in ts:
                t.join()
            
        
        return {'isdone': True}

    def exec_plan(plan, do_log_sub=False):
        # common inputs
        params = {'node': node}
        refined_plan = refine_plan(plan=plan)
        # print('*'*50)
        # print(refined_plan)
        # print('*'*50)
        
        if do_log_sub:
            logger.start()
        
        prev_res = {'isdone': True}
        exec(refined_plan)
                
        if do_log_sub:
            logger.stop()
        
        return prev_res
    
    def actionchunk2actions(action_chunk):
        actions = []
        num_state = len(action_chunk)
        # for i in range(num_state):
        for i in range(num_state):
        # for i in [0,]:
            state = action_chunk[i,:]
            plan = ''
            x,y,z = 1000*state[:3]
            # plan += f'arm_move::{x:.2f},{y:.2f},{z:.2f}\n'
            plan += f'move::x={x},y={y},z={z}'

            # gripper = state[-1]>0.5
            # plan += f'grip::{1000 if gripper else 100}\n'

            actions.append(plan)
        return actions
            


    def exec_prompt(prompt, do_log_sub=False, prompt2='plan'):
        prompt2 = prompt2.strip().lower()
        if 'plan' in prompt2:
            plan = llm_client.chat(prompt=prompt, guide=GUIDE)
            print('='*40)
            print(plan)
            print('='*40)
            
            if prompt2=='plan_only':
                return {'isdone': True}
            
            if do_log_sub:
                logger.start(log_dir= '@' + prompt.replace(' ', '_'))
            ret =  exec_plan(plan, do_log_sub=False)
            if do_log_sub:
                logger.stop()
                time.sleep(1)
                arm_back(node=node)
        elif 'openpi' in prompt2:
            frame = {
                    'image': node.agents['femto_rgb'].get()['im'],
                    'wrist_image' : cv2.rotate(node.agents['dabai_rgb'].get()['im'] , cv2.ROTATE_180),
                    'prompt': prompt}
            frame['state'] = (node.agents['robot_pose'].get()['pose']/1000.).tolist()
            # frame['state'] = np.array(frame['state']  + [int(gripstate_agent.rev_data['motor_position']> 500), ])
            frame['state'] = np.array(frame['state']  + [1, ])

            action_chunk = openpi_client.send({'frame': frame})
            actions=actionchunk2actions(action_chunk=action_chunk)

            if 'value_only' in prompt2:
                ret = {'isdone': True}
            else:
                ret = exec_actions(actions=actions)
        else:
            NotImplementedError
        return ret
        
    
    def taskplan_responsefunc(data):
        if 'plan' in data:
            return exec_plan(plan=data['plan'], do_log_sub=DO_LOG_SUB)
        elif 'actions' in data:
            return exec_actions(actions=data['actions'])
        elif 'prompt' in data:
            exec_prompt(prompt=data['prompt'], do_log_sub=DO_LOG_SUB, prompt2=MANAGER_CONFIGS['prompt2'])
        else:
            return {'isdone': False}

        
    # Deploy node
    node.agents[MANAGER_CONFIGS['conn_name']].response_func = taskplan_responsefunc
    node.listen()


    
if __name__ == '__main__':
    run_node()
