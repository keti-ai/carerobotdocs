def inputs2dict(inputs):
    inp = [el.strip().split('=') for el in inputs.split(',')] if ',' in inputs else [inputs.strip().split('=')]
    return {k: evaluate(v) for k, v in inp}

def grip_encodefunc(data, req):
    inp = data['inputs']
    grip_width = RobotParam.grip_open if inp=='open' else RobotParam.grip_close if inp=='close' else int(inp)
    grip_width = np.clip(grip_width, GRIP_WIDTH_RANGE[0], GRIP_WIDTH_RANGE[1])
    print(f'grip width: {grip_width}')
    set_atrrs(req, {'pose': int(grip_width), 'force': RobotParam.grip_min_force})
    return req

def lift_encode(data, req):
    inp = data['inputs']
    mm = RobotParam.elev_home if inp=='home' else float(inp)
    mm = np.clip(mm, LIFT_RANGE[0], LIFT_RANGE[1])

    print(f'lifting to {mm} mm')
    set_atrrs(req, {'move': float(mm), 'until_complete': True})
    return req

def armdmove_encodefunc(data, req):
    inputs = {'speed': RobotParam.l_arm_speed, 'acc': RobotParam.l_arm_accel, 
              'mvtime': 0.0, 'wait':True, 'relative': True, 'motion_type':0}
    dmoves = [0.,0.,0.,0.,0.,0.]
    dmove_labels = ['dx', 'dy', 'dz', 'drx', 'dry', 'drz']

    inp = inputs2dict(data['inputs'])
    for k, v in inp.items():
        i = dmove_labels.index(k)
        dmoves[i] = float(v) if i<=2 else v/180.*math.pi

    for i in range(3):
        dmoves[i] = np.clip(dmoves[i], DMOVE_RANGE[0], DMOVE_RANGE[1])
    print(f'Arm different moves: {dmoves}')

    set_atrrs(req, update_dict(inputs, {'pose': dmoves}))
    return req

def move_angles_encodefunc(data, req):
    inputs = {'speed': RobotParam.j_arm_speed, 'acc': RobotParam.j_arm_accel}

    inp = data['inputs']
    joint_angles = RobotParam.arm_home if inp=='home' else RobotParam.arm_ready if inp=='ready' else str2joinangle(inp)
    print(f'Join angles: {joint_angles}')

    set_atrrs(req, update_dict(inputs, {'angles': joint_angles,}))
    return req

def movehead_encodefunc(data, req):
    inputs = update_dict({'rz': 0., 'ry':0., 'wait':True}, inputs2dict(data['inputs']))
    inputs['ry'], inputs['rz'] = float(np.clip(inputs['ry'], HEAD_RY_RANGE[0], HEAD_RY_RANGE[1])), float(inputs['rz'])

    set_atrrs(req, inputs)
    return req

def robotpose_decodefunc(msg):
    ret = get_attrs(msg, ['angle', 'pose', 'mode', 'err', 'state'])
    # ret['angle'] = [float(el) for el in ret['angle']]
    # ret['pose'] = [float(el) for el in ret['pose']]
    ret['angle'] = np.array(ret['angle'])
    ret['pose'] = np.array(ret['pose'])
    return ret
