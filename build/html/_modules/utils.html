def node_add_agent_from_config(node, config, agent_name=None, mode=''):  
    if config['agent_type']=='service_server':
        callback_group = node.callback_groups[-1]
    elif config['agent_type']=='sub':
        callback_group = node.callback_groups[-2]
    else:
        callback_group = None
    
    if 'agent_name' not in config:
        config.update({'agent_name': agent_name})
    config['agent_name'] = f'{config["agent_name"]}{mode}'

    node.add_agent(callback_group=callback_group, **config)
    return node

def node_add_agents_from_configs(node, configs, agents_select=None, mode=''):   
    
    for agent_name, cfg in configs.items():
        if agents_select is not None:
            if agent_name not in agents_select:
                continue

        cfg = configs[agent_name]
        
        node = node_add_agent_from_config(node=node, agent_name=agent_name, config=cfg, mode=mode)

    return node

def create_plane(size=10, step=1, noise_level=0.1):
    x = np.arange(-size, size, step)
    y = np.arange(-size, size, step)
    xx, yy = np.meshgrid(x, y)
    zz = np.zeros_like(xx) + np.random.normal(0, noise_level, xx.shape)
    points = np.vstack((xx.ravel(), yy.ravel(), zz.ravel())).T
    return points

def create_parabolic(size=10, step=1, noise_level=0.1):
    x = np.arange(-size, size, step)
    y = np.arange(-size, size, step)
    xx, yy = np.meshgrid(x, y)
    zz = 0.1 * (xx**2 + yy**2) + np.random.normal(0, noise_level, xx.shape)
    points = np.vstack((xx.ravel(), yy.ravel(), zz.ravel())).T
    return points

def calc_normalvector(points):
    centroid = np.mean(points, axis=0)
    centered_points = points - centroid
    cov_matrix = np.cov(centered_points, rowvar=False)
    _, _, vh = np.linalg.svd(cov_matrix)
    normal_vector = vh[-1]  # The last singular vector is the normal
    return normal_vector

def visualize_rgbd_pointcloud(depth, cam_params, rgb=None, normal_point=None, normal_vector=None):
    fx, fy, cx, cy = cam_params
    h, w = depth.shape
    
    # Generate 3D points
    x, y = np.meshgrid(np.arange(w), np.arange(h))
    x3 = (x - cx) * depth / fx
    y3 = (y - cy) * depth / fy
    z3 = depth
    points = np.stack((x3, y3, z3), axis=-1).reshape(-1, 3)
    
    # Create Open3D point cloud
    pcd = o3d.geometry.PointCloud()
    pcd.points = o3d.utility.Vector3dVector(points)
    
    # Assign RGB values as color
    if rgb is not None:
        colors = rgb.reshape(-1, 3) / 255.0
        pcd.colors = o3d.utility.Vector3dVector(colors)

    geometries = [pcd]

    # Add normal vector visualization if provided
    if normal_point is not None and normal_vector is not None:
        line_points = [normal_point, normal_point + 50*normal_vector]
        lines = [[0, 1]]
        line_set = o3d.geometry.LineSet()
        line_set.points = o3d.utility.Vector3dVector(line_points)
        line_set.lines = o3d.utility.Vector2iVector(lines)
        geometries.append(line_set)
    
    # Visualize
    o3d.visualization.draw_geometries(geometries)

def xyz2Ixy(x,y,z, cam_params, eps=1e-10):
    fx, fy, xc, yc = cam_params[:4]
    Ix =  np.divide(x, z + eps) * fx + xc
    Iy =  np.divide(y, z + eps) * fy + yc
    return int(Ix), int(Iy)

def Ixy2xyz(Ix, Iy, Z, cam_params):
    fx, fy, xc, yc = cam_params[:4]
    Ix, Iy = Ix - xc, Iy - yc
    return np.multiply(Ix, Z) / fx, np.multiply(Iy, Z) / fy, Z

    points = np.stack([X.ravel(), Y.ravel(), Z.ravel()], axis=-1)

    return calc_normalvector(points)

def construct_plane(x, y, z):
    # Flatten arrays in case they are multi-dimensional
    x, y, z = x.flatten(), y.flatten(), z.flatten()
    
    # Create matrix A with x, y, and constant term
    A = np.vstack((x, y, np.ones_like(x))).T
    
    # Solve for plane coefficients: ax + by + c = z
    coeffs, _, _, _ = np.linalg.lstsq(A, z, rcond=None)
    return coeffs  # Returns [a, b, c]

def visualize_plane(x, y, z, coeffs):
    import matplotlib.pyplot as plt
    from mpl_toolkits.mplot3d import Axes3D
    fig = plt.figure()
    ax = fig.add_subplot(111, projection='3d')

    # Scatter plot of original points
    ax.scatter(x, y, z, color='r', label='Data points')

    # Create mesh grid for plane
    x_range = np.linspace(np.min(x), np.max(x), 50)
    y_range = np.linspace(np.min(y), np.max(y), 50)
    X, Y = np.meshgrid(x_range, y_range)
    Z = coeffs[0] * X + coeffs[1] * Y + coeffs[2]

    # Plot plane surface
    ax.plot_surface(X, Y, Z, alpha=0.5, color='blue', label='Fitted plane')

    # Labels
    ax.set_xlabel('X')
    ax.set_ylabel('Y')
    ax.set_zlabel('Z')
    ax.set_title('3D Plane Fit')
    # plt.legend()
    plt.show()

def align_pointclouds(source_points, target_points, threshold=0.02, max_iterations=50):
    """
    Aligns two point clouds using the Iterative Closest Point (ICP) algorithm.
    
    Parameters:
        source_points (numpy.ndarray): Nx3 array representing the source point cloud.
        target_points (numpy.ndarray): Nx3 array representing the target point cloud.
        threshold (float): Distance threshold for ICP.
        max_iterations (int): Maximum number of ICP iterations.

    Returns:
        tuple: (transformation matrix (4x4 numpy array), fitness score, RMSE error)
    """
    # Convert numpy arrays to Open3D point clouds
    source_pcd = o3d.geometry.PointCloud()
    target_pcd = o3d.geometry.PointCloud()
    source_pcd.points = o3d.utility.Vector3dVector(source_points)
    target_pcd.points = o3d.utility.Vector3dVector(target_points)
    
    # Pre-align using center of mass to improve translation estimation
    source_center = np.mean(source_points, axis=0)
    target_center = np.mean(target_points, axis=0)
    initial_translation = np.eye(4)
    initial_translation[:3, 3] = target_center - source_center
    
    # Estimate normals (optional, but improves ICP accuracy)
    source_pcd.estimate_normals()
    target_pcd.estimate_normals()
    
    # Run ICP alignment with improved initialization
    icp_result = o3d.pipelines.registration.registration_icp(
        source_pcd, target_pcd, threshold,
        initial_translation,  # Improved initial transformation
        o3d.pipelines.registration.TransformationEstimationPointToPoint(),
        o3d.pipelines.registration.ICPConvergenceCriteria(max_iteration=max_iterations)
    )
    
    transformation = icp_result.transformation
    fitness = icp_result.fitness  # Proportion of inlier correspondences
    rmse = icp_result.inlier_rmse  # Root Mean Square Error of alignment
    
    return transformation, fitness, rmse
